function encontrarPalabraLarga(str)

{

let palabras = str.split(" ");

// console.log(palabras);

let miPalabraLarga = "";

for (let palabra in palabras)

{

if (palabra.length > miPalabraLarga.length)

{

miPalabraLarga = palabra;

}

}

return miPalabraLarga;

}
